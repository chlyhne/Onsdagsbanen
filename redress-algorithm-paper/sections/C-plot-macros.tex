\newcommand{\AddBoatStateSeriesToAxis}[4]{%
\addplot[name path=#4low, draw=none, forget plot] table[x=x_pos,y=x_q25] {#1};
\addplot[name path=#4high, draw=none, forget plot] table[x=x_pos,y=x_q75] {#1};
\addplot[fill=#3, fill opacity=\UncertaintyFillOpacity, draw=none, forget plot] fill between[of=#4high and #4low];
\addplot[solid, mark=none, line width=1.0pt, color=#2, forget plot] table[x=x_pos,y=x_prior] {#1};
\addplot[only marks, mark=*, mark size=1.6pt, opacity=0.95, color=#2, mark options={fill=#2,draw=#2}, forget plot] table[x=x_pos,y=x_obs] {#1};
}
